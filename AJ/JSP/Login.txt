package edu.met.p1;
public class LoginBean {
    String userName;
    String password;
    public LoginBean() {
    }
    public String getUserName() {
        return userName;
    }
    public void setUserName(String userName) {
        this.userName = userName;
    }
    public String getPassword() {
        return password;
    }
    public void setPassword(String password) {
        this.password = password;
    }
    public boolean autenticate()
    {
        if(this.userName.equals("aniketocto") && this.password.equals("aniket123"))
        {
            return true;
        }
        else
        {
            return false;
        }
    }
}

Controller.jsp

<%@ page language="java" contentType="text/html; charset=ISO-8859-1"
 pageEncoding="ISO-8859-1"%>
<!DOCTYPE html>
<html>
<body>
<%
//Get the username and password
String un = request.getParameter("txtUn");
String pwd = request.getParameter("txtPwd");
%>
<!-- Use the login bean -->
<jsp:useBean id="login" class="edu.met.p1.LoginBean" scope="page" />
<jsp:setProperty property="userName" name="login" value="<%=un %>"/>
<jsp:setProperty property="password" name="login" value="<%=pwd
%>"/>
<%
if(login.autenticate())
{
%>
<jsp:forward page="Success.jsp">
<jsp:param value="<%=un %>" name="userName"/>
</jsp:forward>
<%
}
else
{
%>
<jsp:forward page="Failure.jsp" />
<%
}
%>
</body>
</html>

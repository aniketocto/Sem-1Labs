    <tr>
        <th>Movie Id</th>
        <th>Movie Name</th>
        <th>Director</th>
    </tr>

    <c:forEach var="result" items="${myque.rows}">
        <tr>
            <td><c:out value="${result.mid}" /></td>

            <td><c:out value="${result.director}" /></td>
